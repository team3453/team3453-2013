005    package edu.tps.team3453.subsystems;
006    
007    import edu.wpi.first.wpilibj.Encoder;
008    import edu.wpi.first.wpilibj.SpeedController;
009    import edu.wpi.first.wpilibj.Victor;
010    import edu.wpi.first.wpilibj.command.PIDSubsystem;
011    import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
012    import edu.tps.team3453.RobotMap;
013    import edu.tps.team3453.commands.RightDriveMotorStop;
014    
015    /**
016     *
017     * @author digimo
018     */
019    public class RightDriveMotor extends PIDSubsystem {
020    
021        private static final double Kp = 1.0;
022        private static final double Ki = 0.0;
023        private static final double Kd = 0.0;
024        private static final double Kf = -0.11;
025        private static final double KoutMin = -0.3;
026        private static final double KoutMax = -0.08;
027        private static final double Ksetpoint = -550;
028        
029        private static final SpeedController rightMotor = new Victor(RobotMap.rightDriveMotor);
030        private static final Encoder rightEncoder = new Encoder(RobotMap.rightDriveEncoderA,RobotMap.rightDriveEncoderB);
031    
032        // Initialize your subsystem here
033        public RightDriveMotor() {
034            super("RightDriveMotor", Kp, Ki, Kd, Kf);
035    
036            // Use these to get going:
037            // setSetpoint() -  Sets where the PID controller should move the system
038            //                  to
039            // enable() - Enables the PID controller.
040            rightEncoder.setPIDSourceParameter(Encoder.PIDSourceParameter.kRate);
041            
042            disable();
043            rightEncoder.stop();
044            rightEncoder.reset();
045            rightEncoder.start();
046            setSetpoint(Ksetpoint);
047            
048        }
049        
050        public void initDefaultCommand() {
051            // Set the default command for a subsystem here.
052            //setDefaultCommand(new MySpecialCommand());
053            setDefaultCommand(new RightDriveMotorStop());
054        }
055        
056        protected double returnPIDInput() {
057            // Return your input value for the PID loop
058            // e.g. a sensor, like a potentiometer:
059            // yourPot.getAverageVoltage() / kYourMaxVoltage;
060            return rightEncoder.getRate();
061        }
062        
063        protected void usePIDOutput(double output) {
064            // Use output to drive your system, like a motor
065            // e.g. yourMotor.set(output);
066            if (output > KoutMax) {
067                output = KoutMax;
068            }
069            if (output < KoutMin) {
070                output = KoutMin;
071            }
072            rightMotor.set(output);
073        }
074        
075        public void off() {
076            disable();
077            rightMotor.set(0);
078        }
079    
080        public void powerOn(double power) {
081            rightMotor.set(power);
082        }
083        
084        public SpeedController getMotor() {
085            return rightMotor;
086        }
087        
088        public Encoder getEncoder() {
089            return rightEncoder;
090        }
091        
092        public void updateStatus() {
093            SmartDashboard.putNumber("Right Enc count ", rightEncoder.get());
094            SmartDashboard.putNumber("Right Motor out ", rightMotor.get());
095            SmartDashboard.putNumber("Right Enc rate ", rightEncoder.getRate());
096            
097        }    
098    }
