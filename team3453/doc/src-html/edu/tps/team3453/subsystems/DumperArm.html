005    package edu.tps.team3453.subsystems;
006    
007    import edu.wpi.first.wpilibj.command.Subsystem;
008    import edu.wpi.first.wpilibj.SpeedController;
009    import edu.wpi.first.wpilibj.Victor;
010    import edu.tps.team3453.RobotMap;
011    import edu.tps.team3453.Utils;
012    import edu.tps.team3453.commands.DumperArmDoNothing;
013    import edu.wpi.first.wpilibj.DigitalInput;
014    import edu.wpi.first.wpilibj.Relay;
015    
016    /**
017     *
018     * @author Madeline
019     */
020    public class DumperArm extends Subsystem {
021        private static final Relay dumperArmMotor = new Relay(RobotMap.dumperArmMotor);
022        private static final DigitalInput limitSwitchDumperUp = new DigitalInput(2, RobotMap.limitSwitchDumperUp);
023        private static final DigitalInput limitSwitchDumperDown = new DigitalInput(2, RobotMap.limitSwitchDumperDown);
024        public void initDefaultCommand() {
025            setDefaultCommand(new DumperArmDoNothing());
026        }
027        public void Up(){
028            dumperArmMotor.set(Relay.Value.kReverse);
029        }
030        public void Down(){
031            dumperArmMotor.set(Relay.Value.kForward);
032        }
033        public void Stop() {
034            dumperArmMotor.set(Relay.Value.kOff);
035        }    
036        public boolean isUp () {
037            return(Utils.negate(limitSwitchDumperUp.get()));
038        }
039        public boolean isDown () {
040            return(Utils.negate(limitSwitchDumperDown.get()));
041        }
042    }
