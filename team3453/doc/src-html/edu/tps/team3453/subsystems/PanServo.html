005    package edu.tps.team3453.subsystems;
006    
007    import edu.tps.team3453.RobotMap;
008    import edu.tps.team3453.RobotValues;
009    import edu.wpi.first.wpilibj.Servo;
010    import edu.wpi.first.wpilibj.command.Subsystem;
011    import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
012    import edu.tps.team3453.commands.PanServoDoNothing;
013    
014    /**
015     *
016     * @author Madeline
017     */
018    public class PanServo extends Subsystem {
019        Servo servo = new Servo(RobotMap.panServoInput);
020        // Put methods for controlling this subsystem
021        // here. Call these from Commands.
022    
023        public void initDefaultCommand() {
024            // Set the default command for a subsystem here.
025            //setDefaultCommand(new MySpecialCommand());
026            setDefaultCommand(new PanServoDoNothing());
027        }
028        public void panServoDoNothing(){
029            //servo.set(0);
030        }
031        public void panWhileOn(){
032            servo.set(RobotValues.cameraPanServoOnPyramidAngle);
033            SmartDashboard.putNumber("PanServo Degrees", servo.get());
034        }
035        public void panApproach(){
036            servo.set(RobotValues.cameraPanServoPyramidApproachAngle);
037            SmartDashboard.putNumber("PanServo Degrees", servo.get());
038        }
039        public void panServoClockwise(){
040            //servo.set(1);
041            servo.setAngle(servo.getAngle() + 1.0);
042            SmartDashboard.putNumber("PanServo Degrees", servo.getAngle());
043        }
044        public void panServoCounterClockwise(){
045            //servo.set(-1);
046            servo.setAngle(servo.getAngle() - 1.0);
047            SmartDashboard.putNumber("PanServo Degrees", servo.getAngle());
048        }
049    }
