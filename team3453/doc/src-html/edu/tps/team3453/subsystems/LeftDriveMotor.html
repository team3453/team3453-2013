005    package edu.tps.team3453.subsystems;
006    
007    import edu.wpi.first.wpilibj.Encoder;
008    import edu.wpi.first.wpilibj.SpeedController;
009    import edu.wpi.first.wpilibj.Victor;
010    import edu.wpi.first.wpilibj.command.PIDSubsystem;
011    import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
012    import edu.tps.team3453.RobotMap;
013    import edu.tps.team3453.commands.LeftDriveMotorStop;
014    
015    /**
016     *
017     * @author digimo
018     */
019    public class LeftDriveMotor extends PIDSubsystem {
020    
021        private static final double Kp = 1.0;
022        private static final double Ki = 0.0;
023        private static final double Kd = 0.0;
024        private static final double Kf = 0.2;
025        private static final double KoutMin = 0.2;
026        private static final double KoutMax = 0.25;
027        private static final double Ksetpoint = 550;
028        
029        private static final SpeedController leftMotor = new Victor(RobotMap.leftDriveMotor);
030        private static final Encoder leftEncoder = new Encoder(RobotMap.leftDriveEncoderA,RobotMap.leftDriveEncoderB);
031    
032        private static double currentOutput = 0.0;
033        
034        // Initialize your subsystem here
035        public LeftDriveMotor() {
036            super("LeftDriveMotor", Kp, Ki, Kd, Kf);
037    
038            // Use these to get going:
039            // setSetpoint() -  Sets where the PID controller should move the system
040            //                  to
041            // enable() - Enables the PID controller.
042            leftEncoder.setPIDSourceParameter(Encoder.PIDSourceParameter.kRate);
043            
044            disable();
045            leftEncoder.stop();
046            leftEncoder.reset();
047            leftEncoder.start();
048            setSetpoint(Ksetpoint);
049            
050        }
051        
052        public void initDefaultCommand() {
053            // Set the default command for a subsystem here.
054            //setDefaultCommand(new MySpecialCommand());
055            setDefaultCommand(new LeftDriveMotorStop());
056        }
057        
058        protected double returnPIDInput() {
059            // Return your input value for the PID loop
060            // e.g. a sensor, like a potentiometer:
061            // yourPot.getAverageVoltage() / kYourMaxVoltage;
062            return leftEncoder.getRate();
063        }
064        
065        protected void usePIDOutput(double output) {
066            // Use output to drive your system, like a motor
067            // e.g. yourMotor.set(output);
068            if (output > KoutMax) {
069                output = KoutMax;
070            }
071            if (output < KoutMin) {
072                output = KoutMin;
073            }
074            leftMotor.set(output);
075        }
076        
077        public void off() {
078            disable();
079            leftMotor.set(0);
080        }
081    
082        public void powerOn(double power) {
083            leftMotor.set(power);
084        }
085        
086        public SpeedController getMotor() {
087            return leftMotor;
088        }
089        
090        public Encoder getEncoder() {
091            return leftEncoder;
092        }
093        
094        public void updateStatus() {
095            SmartDashboard.putNumber("Left Enc count ", leftEncoder.get());
096            SmartDashboard.putNumber("Left Motor out ", leftMotor.get());
097            SmartDashboard.putNumber("Left Enc rate ", leftEncoder.getRate());
098            
099        }
100    
101        public void setAt10() {
102            System.out.println("left motor set at 10");
103            leftMotor.set(0.1);
104        }
105    
106        public void setAt12() {
107            System.out.println("left motor set at 12");
108            leftMotor.set(0.12);
109        }
110    
111        public void setAt15() {
112            System.out.println("left motor set at 15");
113            leftMotor.set(0.15);
114        }
115    
116        public void setAt20() {
117            System.out.println("left motor set at 20");
118            leftMotor.set(0.2);
119        }
120        
121        public void up5 () {
122            currentOutput += 0.05;
123            checkOutput();
124            runCurrentOutput();
125        }
126        
127        public void down5 () {
128            currentOutput -= 0.05;
129            checkOutput();
130            runCurrentOutput();
131    
132        }
133        
134        public void runCurrentOutput() {
135            System.out.println("Left Motor Current Output is "+currentOutput);
136            leftMotor.set(currentOutput);
137        }
138        
139        private void checkOutput() {
140            if (currentOutput > 1.0) {
141                currentOutput = 1.0;
142            } else if (currentOutput < -1.0) {
143                currentOutput = -1.0;
144            }
145        }
146    }
