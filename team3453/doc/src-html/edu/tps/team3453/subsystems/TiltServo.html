005    package edu.tps.team3453.subsystems;

007    import edu.tps.team3453.RobotMap;
008    import edu.tps.team3453.RobotValues;
009    import edu.wpi.first.wpilibj.Servo;
010    import edu.wpi.first.wpilibj.command.Subsystem;
011    import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
012    import edu.tps.team3453.commands.TiltServoDoNothing;
013    
014    /**
015     *
016     * @author Madeline
017     */
018    public class TiltServo extends Subsystem {
019    Servo servo = new Servo(RobotMap.tiltServoInput);
020        // Put methods for controlling this subsystem
021        // here. Call these from Commands.
022    
023        public void initDefaultCommand() {
024            // Set the default command for a subsystem here.
025            //setDefaultCommand(new MySpecialCommand());
026            setDefaultCommand(new TiltServoDoNothing());
027        }
028        public void tiltServoDoNothing(){
029            //servo.set(0);
030            
031        }
032        public void tiltServoWhileOn(){
033            servo.set(RobotValues.cameraTiltServoOnPyramidAngle);
034            SmartDashboard.putNumber("TiltServo Degrees", servo.get());
035        }
036        public void tiltServoApproach(){
037            servo.set(RobotValues.cameraTiltServoPyramidApproachAngle);
038            SmartDashboard.putNumber("TiltServo Degrees", servo.get());
039        }
040        public void tiltServoClockwise(){
041            servo.setAngle(servo.getAngle() + 1.0);
042            SmartDashboard.putNumber("TiltServo Degrees", servo.getAngle());
043        }
044        public void tiltServoCounterClockwise(){
045            servo.setAngle(servo.getAngle() - 1.0);
046            SmartDashboard.putNumber("TiltServo Degrees", servo.getAngle());
047        }
048       
049    }
