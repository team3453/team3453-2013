005    package edu.tps.team3453.subsystems;
006    
007    import edu.tps.team3453.RobotMap;
008    import edu.tps.team3453.commands.RightArmDoNothing;
009    import edu.wpi.first.wpilibj.DigitalInput;
010    import edu.wpi.first.wpilibj.SpeedController;
011    import edu.wpi.first.wpilibj.Victor;
012    import edu.wpi.first.wpilibj.command.PIDSubsystem;
013    
014    /**
015     *
016     * @author admin
017     */
018    public class RightArm extends PIDSubsystem {
019    
020        private static final double Kp = 0.0;
021        private static final double Ki = 0.0;
022        private static final double Kd = 0.0;
023        
024        SpeedController rightArm = new Victor(RobotMap.rightArm);
025        private static DigitalInput limitSwitchRightArmReach = new DigitalInput(2, RobotMap.limitSwitchRightArmReach);
026        private static DigitalInput limitSwitchRightArmPull = new DigitalInput(2, RobotMap.limitSwitchRightArmPull);
027    
028        private static double currentOutput = 0.0;
029    
030        // Initialize your subsystem here
031        public RightArm() {
032            super("RightArm", Kp, Ki, Kd);
033    
034            // Use these to get going:
035            // setSetpoint() -  Sets where the PID controller should move the system
036            //                  to
037            // enable() - Enables the PID controller.
038        }
039        
040        public void initDefaultCommand() {
041            // Set the default command for a subsystem here.
042            //setDefaultCommand(new MySpecialCommand());
043            setDefaultCommand(new RightArmDoNothing());
044        }
045        
046        protected double returnPIDInput() {
047            // Return your input value for the PID loop
048            // e.g. a sensor, like a potentiometer:
049            // yourPot.getAverageVoltage() / kYourMaxVoltage;
050            return 0.0;
051        }
052        
053        protected void usePIDOutput(double output) {
054            // Use output to drive your system, like a motor
055            // e.g. yourMotor.set(output);
056        }
057        public void rightArmReach(){
058            rightArm.set(-0.2);
059        }
060        public void rightArmPull(){
061            rightArm.set(0.2);
062        }
063        public void stop(){
064            rightArm.set(0);
065        }
066        public boolean isExtended() {
067           return limitSwitchRightArmReach.get();
068        }
069        public boolean isRetracted() {
070            return limitSwitchRightArmPull.get();
071        }
072        
073        public void up5 () {
074            currentOutput -= 0.05;
075            checkOutput();
076            printCurrentOutput();
077        }
078        
079        public void down5 () {
080            currentOutput += 0.05;
081            checkOutput();
082            printCurrentOutput();
083    
084        }
085        
086        public void printCurrentOutput() {
087            System.out.println("Right Arm Current Output is "+currentOutput);
088        }
089        
090        
091        public void runCurrentOutput() {
092            System.out.println("Run Right Arm Current Output is "+currentOutput);
093            rightArm.set(currentOutput);
094        }
095        
096        private void checkOutput() {
097            if (currentOutput > 1.0) {
098                currentOutput = 1.0;
099            } else if (currentOutput < -1.0) {
100                currentOutput = -1.0;
101            }
102        }
103    }
