024    /**
025     * The VM is configured to automatically run this class, and to call the
026     * functions corresponding to each mode, as described in the IterativeRobot
027     * documentation. If you change the name of this class or the package after
028     * creating this project, you must also update the manifest file in the resource
029     * directory.
030     */
031    public class Robot extends IterativeRobot {
032    
033        Command autonomousCommand;
034        Command teleopCommand;
035        /**
036         * This function is run when the robot is first started up and should be
037         * used for any initialization code.
038         */
039        public void robotInit() {
040            // instantiate the command used for the autonomous period
041            autonomousCommand = new AutonomousDrive(14400);
042            teleopCommand = new DriveTeleop();
043            // Initialize all subsystems
044            CommandBase.init();
045        }
046    
047        public void autonomousInit() {
048            // schedule the autonomous command (example)
049            autonomousCommand.start();
050        }
051    
052        /**
053         * This function is called periodically during autonomous
054         */
055        public void autonomousPeriodic() {
056            Scheduler.getInstance().run();
057        }
058    
059        public void teleopInit() {
060            // This makes sure that the autonomous stops running when
061            // teleop starts running. If you want the autonomous to 
062            // continue until interrupted by another command, remove
063            // this line or comment it out.
064            teleopCommand.start();
065        }
066    
067        /**
068         * This function is called periodically during operator control
069         */
070        public void teleopPeriodic() {
071            Scheduler.getInstance().run();
072        }
073        
074        /**
075         * This function is called periodically during test mode
076         */
077        public void testPeriodic() {
078            LiveWindow.run();
079        }
080    }
