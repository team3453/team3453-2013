005    package edu.tps.team3453.commands;
006    
007    /**
008     *
009     * @author digimo
010     */
011    public class AutonomousDrive extends CommandBase {
012        // setpoint is the number of encoder ticks to drive to
013        double setpoint;
014        boolean initrun = true;
015        
016        public AutonomousDrive() {
017            requires(leftDriveMotor);
018            requires(rightDriveMotor);
019        }
020        
021        public AutonomousDrive(double setpoint) {
022            // Use requires() here to declare subsystem dependencies
023            // eg. requires(chassis);
024            requires(leftDriveMotor);
025            requires(rightDriveMotor);
026            this.setpoint = setpoint;
027        }
028    
029        // Called just before this Command runs the first time
030        protected void initialize() {
031        }
032    
033        // Called repeatedly when this Command is scheduled to run
034        protected void execute() {
035            if (initrun) {
036                leftDriveMotor.getEncoder().reset();
037                leftDriveMotor.getEncoder().start();
038                rightDriveMotor.getEncoder().reset();
039                rightDriveMotor.getEncoder().start();
040                leftDriveMotor.enable();
041                rightDriveMotor.enable();
042                initrun = false;
043            }
044            leftDriveMotor.updateStatus();
045            rightDriveMotor.updateStatus();
046        }
047    
048        // Make this return true when this Command no longer needs to run execute()
049        protected boolean isFinished() {
050            double leftDiff = Math.abs(Math.abs(leftDriveMotor.getEncoder().get()) - setpoint);
051            double rightDiff = Math.abs(Math.abs(rightDriveMotor.getEncoder().get()) - setpoint);
052            // if either are within half an encoder turn
053            if ((leftDiff < 500) || (rightDiff < 500)) {
054                initrun = true;
055                return true;
056            } else {
057                return false;
058            }
059        }
060    
061        // Called once after isFinished returns true
062        protected void end() {
063            initrun = true;
064            leftDriveMotor.off();
065            rightDriveMotor.off();
066        }
067    
068        // Called when another command which requires one or more of the same
069        // subsystems is scheduled to run
070        protected void interrupted() {
071            initrun = true;
072            leftDriveMotor.off();
073            rightDriveMotor.off();
074        }
075        
076        public void setSetPoint (double setpoint) {
077            this.setpoint = setpoint;
078        }
079    }
