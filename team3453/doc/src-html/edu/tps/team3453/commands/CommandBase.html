020    /**
021     * The base for all commands. All atomic commands should subclass CommandBase.
022     * CommandBase stores creates and stores each control system. To access a
023     * subsystem elsewhere in your code in your code use CommandBase.exampleSubsystem
024     * @author Author
025     */
026    public abstract class CommandBase extends Command {
027    
028        public static OI oi;
029        // Create a single static instance of all of your subsystems
030        public static final LeftDriveMotor leftDriveMotor = new LeftDriveMotor();
031        public static final RightDriveMotor rightDriveMotor = new RightDriveMotor();
032        public static final Lid lid = new Lid();
033        public static final DumperArm dumperArm = new DumperArm();
034        public static final TiltServo tiltServo = new TiltServo();
035        public static final PanServo panServo = new PanServo();
036        public static final LeftArm leftArm = new LeftArm();
037        public static final RightArm rightArm = new RightArm();
038        public static final LeftSolenoid leftSolenoid = new LeftSolenoid();
039        public static final RightSolenoid rightSolenoid = new RightSolenoid();
040        public static final ClimberChassis climberChassis = new ClimberChassis();
041        public static final RearWheel rearWheel = new RearWheel();
042        
043        public static final TestLimitSwitch testLimitSwitch = new TestLimitSwitch();
044        
045        public static void init() {
046            // This MUST be here. If the OI creates Commands (which it very likely
047            // will), constructing it during the construction of CommandBase (from
048            // which commands extend), subsystems are not guaranteed to be
049            // yet. Thus, their requires() statements may grab null pointers. Bad
050            // news. Don't move it.
051            oi = new OI();
052    
053            // Show what command your subsystem is running on the SmartDashboard
054            //updateStatus();
055        }
056    
057        public CommandBase(String name) {
058            super(name);
059        }
060    
061        public CommandBase() {
062            super();
063        }
064        
065        public static void updateStatus() {
066            SmartDashboard.putData(leftDriveMotor);
067            SmartDashboard.putData(rightDriveMotor);
068            
069        }
070    }
